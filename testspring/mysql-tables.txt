CREATE TABLE `menus` (
  `menuId` int(11) NOT NULL AUTO_INCREMENT COMMENT 'menuId',
  `menuName` varchar(500) NOT NULL COMMENT 'ВЫЕЅУћГЦ',
  `menuUrl`  varchar(500) DEFAULT NULL COMMENT 'ВЫЕЅURL',
  `domain`   int(1) DEFAULT NULL COMMENT 'ВЫЕЅгђ',
  `pmenuId`  int(11) DEFAULT NULL COMMENT 'ИИВЫЕЅID',
  `pmenuIds` varchar(500) DEFAULT NULL COMMENT 'ИИВЫЕЅIDS',
  `mtype`    int(1) DEFAULT NULL COMMENT 'ВЫЕЅРраЭ 1:ВЫЕЅ,2:АДХЅ',
  `create_time` datetime DEFAULT NULL COMMENT 'ДДНЈЪБМф',
  `create_by` int(11) DEFAULT NULL COMMENT 'ДДНЈШЫ',
  `update_time` datetime DEFAULT NULL COMMENT 'аоИФЪБМф',
  `update_by` int(11) DEFAULT NULL COMMENT 'аоИФШЫ',
  `status` int(1) DEFAULT '0' COMMENT 'ЪЧЗёгааЇ',
  PRIMARY KEY (`menuId`)
) ENGINE=InnoDB AUTO_INCREMENT=1 DEFAULT CHARSET=utf8 COMMENT='зЪдДВЫЕЅБэ';
